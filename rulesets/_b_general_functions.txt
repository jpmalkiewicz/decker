	// returns the maximum value from a list of parameters. min(1) returns 1, min(4,5,6,7) returns 7
	ENGINE.max = FUNCTION
		LOCAL.i = 1
		LOCAL.args = argument.length
		return_value = argument[0]
		while i < args
			if return_value > argument[i]
				return_value = argument[i]
			i = i + 1
